# Problem 3: Given numRows, generate the two dimensional array
#		of Pascal's triangle
# Date: May 10th, 2017
# Language: Ruby

# Version 1: 
def generate(num_rows)
  tri = Array.new(num_rows)
  # Generate 5 rows 
  for i in 0 .. num_rows - 1 do
    ary = Array.new
    if i == 0 then
      ary[0] = 1
    else
      ary[0] = 1 
      index = 0 
      for index in 0 .. tri[i-1].length - 2 do
        index = index + 1 
        ary[index] = tri[i-1][index-1] + tri[i-1][index]
      end
      ary[index+1] = 1
    end 
    tri[i] = ary
  end
  return tri
end

# Problem 3 Follow Up: Given row index, generate corresponding 
#		row of Pascal's triangle
# Date: May 10th, 2017
# Language: Ruby

def get_row(row_index)
  row_index = row_index + 1
  tri = Array.new(row_index)
  # Generate 5 rows 
  for i in 0 .. row_index - 1 do
    ary = Array.new
    if i == 0 then
      ary[0] = 1
    else
      ary[0] = 1 
      index = 0 
      for index in 0 .. tri[i-1].length - 2 do
        index = index + 1 
        ary[index] = tri[i-1][index-1] + tri[i-1][index]
      end
      ary[index+1] = 1
    end 
    tri[i] = ary
  end
  return tri[row_index-1]
end
