# Problem 4: Find if an array contains duplicate number
#		If yes, return true. Otherwise, return false
# Date: May 10th, 2017
# Language: Ruby

# Version 1: Hash Table -- Time Complexity O(n), Space Complexity O(n)
def contains_duplicate(nums)
    track = Hash.new("")
    for i in 0 .. nums.length - 1 do
        if track.has_value?(nums[i]) then
            return true
        else
            track[i] = nums[i];
        end
    end
    return false
end

# Version 2: Nested For Loop -- Time Complexity O(n^2), Space Complexity: O(1)
def contains_duplicate(nums)
    for i in 0 .. nums.length - 1 do 
      for j in i + 1 .. nums.length - 1 do
        if nums[i] == nums[j] then
          return true 
        end
      end 
    end 
    return false
end
